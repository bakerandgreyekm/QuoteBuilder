const PROJECT_WORKBOOK_ID   = '1zZQOmWe02oF7awuRkqgsMsJU1GwgcINyaXBNfUADkS4';
const REFERENCE_WORKBOOK_ID = '1tdLzOTnnXIVVKrgEgQtS5YVnsZT3Sepo1E4LcKuRR6Y';
const CATALOGUE_WORKBOOK_ID = '1CnwJR9CvDv2TV47dAPGLx-hwZUcEsnXgYYGHVaRB7ts';
const API_KEY               = 'BG_QUOTE_2026';
const PROJECTS_SHEET_NAME   = 'Projects';
const SYSTEM_TYPES_SHEET_NAME = 'SystemTypes';

function checkAuth(e) {
  return e.parameter['key'] === API_KEY;
}

function sendJSON(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function sheetToJSON(sheet) {
  const data = sheet.getDataRange().getValues();
  if (data.length < 2) return [];
  const headers = data[0];
  const rows = [];
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (row.every(cell => cell === '' || cell === null || cell === undefined)) continue;
    const obj = {};
    headers.forEach((header, j) => { obj[header] = row[j]; });
    rows.push(obj);
  }
  return rows;
}

function generateRefNumber(projectsSheet) {
  const data = projectsSheet.getDataRange().getValues();
  if (data.length < 2) return 'BG-QT-1000';
  const refNumbers = data.slice(1).map(row => row[0]).filter(r => /^BG-QT-\d+$/.test(String(r)));
  if (refNumbers.length === 0) return 'BG-QT-1000';
  const highest = Math.max(...refNumbers.map(r => parseInt(String(r).replace('BG-QT-', ''))));
  return 'BG-QT-' + (highest + 1);
}

function generateId() {
  return Utilities.getUuid();
}

function doGet(e) {
  if (!checkAuth(e)) {
    return sendJSON({ success: false, error: 'Unauthorized' });
  }
  const action = e.parameter.action;

  if (action === 'getProjects') {
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(PROJECTS_SHEET_NAME);
      if (!sheet) return sendJSON({ success: false, error: 'Projects sheet not found' });
      return sendJSON({ success: true, data: sheetToJSON(sheet) });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'getSystemTypes') {
    try {
      const wb = SpreadsheetApp.openById(REFERENCE_WORKBOOK_ID);
      const sheet = wb.getSheetByName(SYSTEM_TYPES_SHEET_NAME);
      if (!sheet) return sendJSON({ success: false, error: 'SystemTypes sheet not found' });
      return sendJSON({ success: true, data: sheetToJSON(sheet) });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  // Returns all products across every sheet in the Catalogue workbook.
  // Each row gets Category injected from its sheet name.
  if (action === 'getCatalogue') {
    try {
      const wb = SpreadsheetApp.openById(CATALOGUE_WORKBOOK_ID);
      const rows = [];
      for (const sheet of wb.getSheets()) {
        const category = sheet.getName();
        sheetToJSON(sheet).forEach(r => {
          rows.push({ ...r, Category: category, Rate: parseFloat(r['Rate']) || 0 });
        });
      }
      return sendJSON({ success: true, data: rows });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  // Returns products for a single category (one sheet fetch — faster than getCatalogue).
  if (action === 'getCatalogueByCategory') {
    const category = e.parameter.category;
    if (!category) return sendJSON({ success: false, error: 'category required' });
    try {
      const wb = SpreadsheetApp.openById(CATALOGUE_WORKBOOK_ID);
      const sheet = wb.getSheetByName(category);
      if (!sheet) return sendJSON({ success: false, error: `Category sheet "${category}" not found` });
      const rows = sheetToJSON(sheet).map(r => ({
        ...r, Category: category, Rate: parseFloat(r['Rate']) || 0
      }));
      return sendJSON({ success: true, data: rows });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'getLineItems') {
    const refNumber = e.parameter.refNumber;
    if (!refNumber) return sendJSON({ success: false, error: 'refNumber required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: false, error: 'Project sheet not found' });
      const rows = sheetToJSON(sheet)
        .filter(r => !String(r['ID'] || '').startsWith('SYSTEM_PLACEHOLDER_'));
      const parsed = rows.map(r => ({
        'ID': String(r['ID'] || ''),
        'System Type': String(r['System Type'] || ''),
        'Category': String(r['Category'] || ''),
        'Product Name': String(r['Product Name'] || ''),
        'Brand': String(r['Brand'] || ''),
        'Unit': String(r['Unit'] || ''),
        'Quantity': parseFloat(r['Quantity']) || 0,
        'Rate': parseFloat(r['Rate']) || 0,
        'Amount': parseFloat(r['Amount']) || 0,
        'Note': String(r['Note'] || ''),
        'Created At': String(r['Created At'] || ''),
        'Area': String(r['Area'] || ''),
      }));
      return sendJSON({ success: true, data: parsed });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'getSystems') {
    const refNumber = e.parameter.refNumber;
    if (!refNumber) return sendJSON({ success: false, error: 'refNumber required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: true, data: [] });
      const rows = sheetToJSON(sheet);
      const seen = new Set();
      const systemTypes = [];
      rows.forEach(r => {
        const t = String(r['System Type'] || '');
        if (t && !seen.has(t)) { seen.add(t); systemTypes.push({ 'System Type': t }); }
      });
      return sendJSON({ success: true, data: systemTypes });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'getEmployees') {
    try {
      const wb = SpreadsheetApp.openById(REFERENCE_WORKBOOK_ID);
      const sheet = wb.getSheetByName('Employees');
      if (!sheet) return sendJSON({ success: false, error: 'Employees sheet not found' });
      const values = sheet.getDataRange().getValues().slice(1);
      const names = values.map(r => String(r[0]).trim()).filter(n => n !== '');
      return sendJSON({ success: true, data: names.map(n => ({ Names: n })) });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  return sendJSON({ success: false, error: 'Unknown action' });
}

function doPost(e) {
  let body;
  try { body = JSON.parse(e.postData.contents); }
  catch (err) { return sendJSON({ success: false, error: 'Invalid JSON body' }); }

  if (body.key !== API_KEY) {
    return sendJSON({ success: false, error: 'Unauthorized' });
  }
  const action = body.action;

  if (action === 'createProject') {
    const { projectName, clientName, location, worker } = body;
    if (!projectName) return sendJSON({ success: false, error: 'projectName required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      let projectsSheet = wb.getSheetByName(PROJECTS_SHEET_NAME);
      if (!projectsSheet) {
        projectsSheet = wb.insertSheet(PROJECTS_SHEET_NAME);
        projectsSheet.appendRow(['Ref. Number', 'Project Name', 'Client Name', 'Location', 'Created At', 'Worker', 'Industry', 'Tier', 'Areas']);
      }
      const refNumber = generateRefNumber(projectsSheet);
      const headers = projectsSheet.getRange(1, 1, 1, projectsSheet.getLastColumn()).getValues()[0];
      const newRow = new Array(headers.length).fill('');
      const set = (name, value) => { const i = headers.indexOf(name); if (i >= 0) newRow[i] = value; };
      set('Ref. Number', refNumber);
      set('Project Name', projectName);
      set('Client Name', clientName || '');
      set('Location', location || '');
      set('Created At', new Date().toISOString());
      set('Worker', worker || '');
      set('Industry', body.industry || '');
      set('Tier', body.tier || '');
      set('Areas', '');
      projectsSheet.appendRow(newRow);
      const newSheet = wb.insertSheet(refNumber);
      newSheet.appendRow(['ID', 'System Type', 'Category', 'Product Name', 'Brand', 'Unit', 'Quantity', 'Rate', 'Amount', 'Note', 'Created At', 'Worker', 'Area']);
      const headerRange = newSheet.getRange(1, 1, 1, newSheet.getLastColumn());
      headerRange.setBackground('#000000');
      headerRange.setFontColor('#FFFFFF');
      headerRange.setFontWeight('bold');
      return sendJSON({ success: true, refNumber: refNumber });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'addLineItem') {
    const refNumber = body.refNumber;
    if (!refNumber) return sendJSON({ success: false, error: 'refNumber required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: false, error: 'Project sheet not found' });
      const id = generateId();
      const now = new Date().toISOString();
      const amount = (parseFloat(body.quantity) || 0) * (parseFloat(body.rate) || 0);
      const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      const newRow = new Array(headers.length).fill('');
      const set = (name, value) => { const i = headers.indexOf(name); if (i >= 0) newRow[i] = value; };
      set('ID', id);
      set('System Type', body.systemType || '');
      set('Category', body.category || '');
      set('Product Name', body.productName || '');
      set('Brand', body.brand || '');
      set('Unit', body.unit || '');
      set('Quantity', parseFloat(body.quantity) || 0);
      set('Rate', parseFloat(body.rate) || 0);
      set('Amount', amount);
      set('Note', body.noteText || '');
      set('Created At', now);
      set('Worker', body.worker || '');
      set('Area', body.area || '');
      sheet.appendRow(newRow);
      return sendJSON({ success: true, id: id });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'deleteLineItem') {
    const { refNumber, itemId } = body;
    if (!refNumber || !itemId) return sendJSON({ success: false, error: 'Missing required fields' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: false, error: 'Project sheet not found' });
      const data = sheet.getDataRange().getValues();
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === String(itemId)) {
          sheet.deleteRow(i + 1);
          return sendJSON({ success: true });
        }
      }
      return sendJSON({ success: false, error: 'Item not found' });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'addSystem') {
    const { refNumber, systemType } = body;
    if (!refNumber || !systemType) return sendJSON({ success: false, error: 'Missing required fields' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: false, error: 'Project sheet not found' });
      const placeholderId = 'SYSTEM_PLACEHOLDER_' + generateId();
      sheet.appendRow([placeholderId, systemType, '', '', '', '', '', '', '', '', new Date().toISOString(), '']);
      return sendJSON({ success: true });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'removeSystem') {
    const { refNumber, systemType } = body;
    if (!refNumber || !systemType) return sendJSON({ success: false, error: 'Missing required fields' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: false, error: 'Project sheet not found' });
      const data = sheet.getDataRange().getValues();
      let deletedCount = 0;
      for (let i = data.length - 1; i >= 1; i--) {
        if (String(data[i][1]) === systemType) {
          sheet.deleteRow(i + 1);
          deletedCount++;
        }
      }
      return sendJSON({ success: true, deletedCount: deletedCount });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'updateProject') {
    const { refNumber, projectName, clientName, location } = body;
    if (!refNumber) return sendJSON({ success: false, error: 'refNumber required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(PROJECTS_SHEET_NAME);
      if (!sheet) return sendJSON({ success: false, error: 'Projects sheet not found' });
      const data = sheet.getDataRange().getValues();
      const headers = data[0];
      const industryCol = headers.indexOf('Industry') + 1;
      const tierCol = headers.indexOf('Tier') + 1;
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === refNumber) {
          if (projectName !== undefined) sheet.getRange(i + 1, 2).setValue(projectName);
          if (clientName !== undefined) sheet.getRange(i + 1, 3).setValue(clientName);
          if (location !== undefined) sheet.getRange(i + 1, 4).setValue(location);
          if (industryCol > 0) sheet.getRange(i + 1, industryCol).setValue(body.industry || '');
          if (tierCol > 0) sheet.getRange(i + 1, tierCol).setValue(body.tier || '');
          return sendJSON({ success: true });
        }
      }
      return sendJSON({ success: false, error: 'Project not found' });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'updateProjectAreas') {
    const { refNumber, areas } = body;
    if (!refNumber) return sendJSON({ success: false, error: 'refNumber required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(PROJECTS_SHEET_NAME);
      if (!sheet) return sendJSON({ success: false, error: 'Projects sheet not found' });
      const data = sheet.getDataRange().getValues();
      const headers = data[0];
      const areasCol = headers.indexOf('Areas') + 1;
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === refNumber) {
          if (areasCol > 0) sheet.getRange(i + 1, areasCol).setValue(areas || '');
          return sendJSON({ success: true });
        }
      }
      return sendJSON({ success: false, error: 'Project not found' });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'updateLineItem') {
    const { refNumber, itemId } = body;
    if (!refNumber || !itemId) return sendJSON({ success: false, error: 'Missing required fields' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const sheet = wb.getSheetByName(refNumber);
      if (!sheet) return sendJSON({ success: false, error: 'Project sheet not found' });
      const data = sheet.getDataRange().getValues();
      const headers = data[0];
      for (let i = 1; i < data.length; i++) {
        if (String(data[i][0]) === String(itemId)) {
          const qtyCol = headers.indexOf('Quantity') + 1;
          const rateCol = headers.indexOf('Rate') + 1;
          const amountCol = headers.indexOf('Amount') + 1;
          const noteCol = headers.indexOf('Note') + 1;
          const areaCol = headers.indexOf('Area') + 1;
          if (body.quantity !== undefined && qtyCol > 0) {
            const qty = parseFloat(body.quantity) || 0;
            const rate = rateCol > 0 ? (parseFloat(data[i][rateCol - 1]) || 0) : 0;
            sheet.getRange(i + 1, qtyCol).setValue(qty);
            if (amountCol > 0) sheet.getRange(i + 1, amountCol).setValue(qty * rate);
          }
          if (body.noteText !== undefined && noteCol > 0) sheet.getRange(i + 1, noteCol).setValue(body.noteText);
          if (body.area !== undefined && areaCol > 0) sheet.getRange(i + 1, areaCol).setValue(body.area);
          if (body.rate !== undefined && rateCol > 0) {
            const newRate = parseFloat(body.rate) || 0;
            const qty = qtyCol > 0 ? (parseFloat(data[i][qtyCol - 1]) || 0) : 0;
            sheet.getRange(i + 1, rateCol).setValue(newRate);
            if (amountCol > 0) sheet.getRange(i + 1, amountCol).setValue(qty * newRate);
          }
          return sendJSON({ success: true });
        }
      }
      return sendJSON({ success: false, error: 'Item not found' });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  if (action === 'deleteProject') {
    const { refNumber } = body;
    if (!refNumber) return sendJSON({ success: false, error: 'refNumber required' });
    try {
      const wb = SpreadsheetApp.openById(PROJECT_WORKBOOK_ID);
      const projectSheet = wb.getSheetByName(refNumber);
      if (projectSheet) wb.deleteSheet(projectSheet);
      const projectsIndex = wb.getSheetByName(PROJECTS_SHEET_NAME);
      if (projectsIndex) {
        const data = projectsIndex.getDataRange().getValues();
        for (let i = data.length - 1; i >= 1; i--) {
          if (String(data[i][0]) === refNumber) {
            projectsIndex.deleteRow(i + 1);
            break;
          }
        }
      }
      return sendJSON({ success: true });
    } catch (err) { return sendJSON({ success: false, error: err.message }); }
  }

  return sendJSON({ success: false, error: 'Unknown action' });
}

// Run this ONCE from the Apps Script editor to wire up all catalogue triggers.
// Select "installCatalogueTriggers" in the function dropdown and click ▶ Run.
function installCatalogueTriggers() {
  const ss = SpreadsheetApp.openById(CATALOGUE_WORKBOOK_ID);

  // Remove any existing catalogue triggers first to avoid duplicates
  ScriptApp.getProjectTriggers()
    .filter(t => ['onCatalogueEdit', 'onCatalogueChange', 'onCatalogueOpen']
      .includes(t.getHandlerFunction()))
    .forEach(t => ScriptApp.deleteTrigger(t));

  ScriptApp.newTrigger('onCatalogueEdit').forSpreadsheet(ss).onEdit().create();
  ScriptApp.newTrigger('onCatalogueChange').forSpreadsheet(ss).onChange().create();
  ScriptApp.newTrigger('onCatalogueOpen').forSpreadsheet(ss).onOpen().create();

  Logger.log('Catalogue triggers installed successfully.');
}

// Adds QuoteBuilder menu when the Catalogue workbook is opened.
function onCatalogueOpen() {
  SpreadsheetApp.getUi()
    .createMenu('QuoteBuilder')
    .addItem('Fill Missing IDs', 'fillMissingIds')
    .addItem('Validate Catalogue', 'validateCatalogue')
    .addToUi();
}

// Fires on structural changes (new sheet, inserted rows, paste from another file).
// When a new sheet tab is created, auto-adds the header row.
function onCatalogueChange(e) {
  if (e.changeType !== 'INSERT_GRID') return;
  const sheet = e.source.getActiveSheet();
  if (!sheet || sheet.getLastRow() > 0) return; // already has content
  sheet.appendRow(['ID', 'Product Name', 'Brand', 'Unit', 'Rate', 'Tier']);
  const header = sheet.getRange(1, 1, 1, 6);
  header.setBackground('#000000');
  header.setFontColor('#FFFFFF');
  header.setFontWeight('bold');
}

// Fires on edit/paste in the Catalogue workbook.
// ID is generated as soon as a row gets a non-empty Product Name (col 2).
// Handles single-cell edits and multi-row pastes.
function onCatalogueEdit(e) {
  const sheet = e.range.getSheet();
  const firstDataRow = Math.max(e.range.getRow(), 2); // skip header row
  const lastRow = e.range.getLastRow();
  if (firstDataRow > lastRow) return;

  const category = sheet.getName();

  // Read current sheet state once (post-edit values already reflected here)
  const allData = sheet.getDataRange().getValues();

  // Build prefix state from all existing IDs so new IDs don't clash
  const categoryPrefixes = {};
  const prefixMaxNum = {};
  for (let i = 1; i < allData.length; i++) {
    if (allData[i][0]) _absorbRow(allData[i][0], category, categoryPrefixes, prefixMaxNum);
  }

  // Columns: ID(0) | Product Name(1) | Brand(2) | Unit(3) | Rate(4) | Tier(5)
  for (let row = firstDataRow; row <= lastRow; row++) {
    const r = allData[row - 1];
    if (!r) continue;
    if (r[0]) continue;                                  // already stamped
    if (!String(r[1] || '').trim()) continue;            // no product name yet
    const newId = _generateId(category, categoryPrefixes, prefixMaxNum);
    sheet.getRange(row, 1).setValue(newId);
    _absorbRow(newId, category, categoryPrefixes, prefixMaxNum);
  }
}

// Run from the QuoteBuilder menu to stamp IDs on all rows missing one.
function fillMissingIds() {
  const wb = SpreadsheetApp.openById(CATALOGUE_WORKBOOK_ID);
  const sheets = wb.getSheets();
  let filled = 0;

  for (const sheet of sheets) {
    const category = sheet.getName();
    const data = sheet.getDataRange().getValues();
    if (data.length < 2) continue;

    const categoryPrefixes = {};
    const prefixMaxNum = {};

    // First pass: absorb existing IDs to know what prefixes are taken
    for (let i = 1; i < data.length; i++) {
      _absorbRow(data[i][0], category, categoryPrefixes, prefixMaxNum);
    }

    // Second pass: fill blanks — any row with a Product Name but no ID
    for (let i = 1; i < data.length; i++) {
      if (data[i].every(c => c === '' || c === null)) continue;
      if (data[i][0]) continue;
      if (!String(data[i][1] || '').trim()) continue;
      const id = _generateId(category, categoryPrefixes, prefixMaxNum);
      sheet.getRange(i + 1, 1).setValue(id);
      filled++;
    }

    _sortSheet(sheet);
  }

  SpreadsheetApp.getUi().alert(`Done. Filled ${filled} missing ID${filled !== 1 ? 's' : ''}.`);
}

// Validates all sheets in the Catalogue workbook and checks SystemTypes tags match.
function validateCatalogue() {
  const catWb = SpreadsheetApp.openById(CATALOGUE_WORKBOOK_ID);
  const sheets = catWb.getSheets();
  const knownCategories = new Set(sheets.map(s => s.getName()));
  const issues = [];

  for (const sheet of sheets) {
    const sheetName = sheet.getName();
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i].every(c => c === '' || c === null)) continue;
      const [id, productName, brand, unit, rate] = data[i];
      if (!productName) issues.push(`${sheetName} row ${i + 1}: missing Product Name`);
      if (!brand)       issues.push(`${sheetName} row ${i + 1}: missing Brand`);
      if (!unit)        issues.push(`${sheetName} row ${i + 1}: missing Unit`);
      if (rate === '' || rate === null || rate === undefined)
                        issues.push(`${sheetName} row ${i + 1}: missing Rate`);
    }
  }

  // Check that every SystemTypes tag exists as a sheet in the Catalogue workbook
  try {
    const refWb = SpreadsheetApp.openById(REFERENCE_WORKBOOK_ID);
    const sysSheet = refWb.getSheetByName(SYSTEM_TYPES_SHEET_NAME);
    if (sysSheet) {
      const sysData = sysSheet.getDataRange().getValues();
      for (let i = 1; i < sysData.length; i++) {
        if (sysData[i].every(c => c === '' || c === null)) continue;
        const systemType = String(sysData[i][0] || '').trim();
        const tags = String(sysData[i][1] || '').trim();
        if (!tags) continue;
        tags.split(',').forEach(tag => {
          const t = tag.trim();
          if (t && !knownCategories.has(t)) {
            issues.push(`SystemTypes row ${i + 1} (${systemType}): tag "${t}" not found as a sheet in Catalogue workbook`);
          }
        });
      }
    }
  } catch (_) {}

  if (issues.length === 0) {
    SpreadsheetApp.getUi().alert('✓ All good! No issues found.');
  } else {
    const preview = issues.slice(0, 20).join('\n');
    const extra = issues.length > 20 ? `\n\n...and ${issues.length - 20} more.` : '';
    SpreadsheetApp.getUi().alert(`Found ${issues.length} issue${issues.length !== 1 ? 's' : ''}:\n\n${preview}${extra}`);
  }
}

// Helpers

function _sortSheet(sheet) {
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  if (lastRow <= 1) return;
  // Sort by ID (col A)
  sheet.getRange(2, 1, lastRow - 1, lastCol).sort([{ column: 1, ascending: true }]);
}

function _absorbRow(id, category, categoryPrefixes, prefixMaxNum) {
  id = String(id || '').trim();
  category = String(category || '').trim();
  if (!id || !category) return;
  const match = id.match(/^([A-Z]+)-(\d+)$/);
  if (!match) return;
  const prefix = match[1];
  const num = parseInt(match[2]);
  if (!categoryPrefixes[category]) categoryPrefixes[category] = prefix;
  if (!prefixMaxNum[prefix] || num > prefixMaxNum[prefix]) prefixMaxNum[prefix] = num;
}

// Builds prefix state from all rows in a sheet (used by onCatalogueEdit).
function _buildCatalogueState(sheet, skipSheetRow) {
  const data = sheet.getDataRange().getValues();
  const category = sheet.getName();
  const categoryPrefixes = {};
  const prefixMaxNum = {};
  for (let i = 1; i < data.length; i++) {
    if (i + 1 === skipSheetRow) continue;
    _absorbRow(data[i][0], category, categoryPrefixes, prefixMaxNum);
  }
  return { categoryPrefixes, prefixMaxNum };
}

function _generateId(category, categoryPrefixes, prefixMaxNum) {
  if (categoryPrefixes[category]) {
    const prefix = categoryPrefixes[category];
    const next = (prefixMaxNum[prefix] || 0) + 1;
    prefixMaxNum[prefix] = next;
    return `${prefix}-${String(next).padStart(3, '0')}`;
  }
  const usedPrefixes = new Set(Object.values(categoryPrefixes));
  const prefix = _derivePrefix(category, usedPrefixes);
  categoryPrefixes[category] = prefix;
  prefixMaxNum[prefix] = 1;
  return `${prefix}-001`;
}

// Derives a unique 2-3 letter prefix from category initials.
function _derivePrefix(category, usedPrefixes) {
  const words = category
    .replace(/&/g, 'and')
    .replace(/[^a-zA-Z0-9\s]/g, ' ')
    .trim()
    .split(/\s+/)
    .filter(w => w.length > 0);

  if (words.length === 0) return 'X';

  const contribs = new Array(words.length).fill(1);

  for (let attempt = 0; attempt < 50; attempt++) {
    const prefix = words
      .map((w, i) => w.substring(0, contribs[i]).toUpperCase())
      .join('');

    if (!usedPrefixes.has(prefix)) return prefix;

    let expanded = false;
    for (let pass = 0; pass < words.length; pass++) {
      const idx = (attempt + pass) % words.length;
      if (contribs[idx] < words[idx].length) {
        contribs[idx]++;
        expanded = true;
        break;
      }
    }
    if (!expanded) break;
  }

  const base = words.map(w => w[0].toUpperCase()).join('');
  return base + String(usedPrefixes.size + 1).padStart(2, '0');
}
